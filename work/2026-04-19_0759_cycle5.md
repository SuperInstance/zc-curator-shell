#!/usr/bin/env python3
"""
Categorize 20 uncategorized SuperInstance repos using inferred fleet taxonomy.
Uses GitHub API (public endpoints).
"""
import requests
import json
from typing import List, Dict
import time

# GitHub org
ORG = "SuperInstance"
API_URL = f"https://api.github.com/orgs/{ORG}/repos"

def fetch_repos() -> List[Dict]:
    """Fetch public repos from SuperInstance org."""
    repos = []
    page = 1
    while len(repos) < 50:  # limit to 50 for speed
        resp = requests.get(f"{API_URL}?per_page=100&page={page}")
        if resp.status_code != 200:
            print(f"Error fetching repos: {resp.status_code}")
            break
        batch = resp.json()
        if not batch:
            break
        repos.extend(batch)
        page += 1
        time.sleep(0.5)  # rate limit courtesy
    return repos

def categorize(repo: Dict) -> str:
    """Assign a category based on name, description, topics."""
    name = repo["name"].lower()
    desc = (repo["description"] or "").lower()
    topics = repo.get("topics", [])

    # Check topics first
    topic_map = {
        "plato": "training-rooms",
        "holodeck": "core-infra",
        "simulator": "simulation",
        "rust": "core-infra",
        "torch": "core-infra",
        "agent": "agent-runtime",
        "lora": "ensigns",
        "research": "research",
        "docs": "docs",
        "tool": "tools",
    }
    for topic in topics:
        for key, cat in topic_map.items():
            if key in topic.lower():
                return cat

    # Name/description heuristics
    if "plato" in name or "room" in name:
        return "training-rooms"
    if "holodeck" in name or "flux" in name:
        return "core-infra"
    if "simulator" in name or "constraint" in name:
        return "simulation"
    if "cudaclaw" in name or "zeroclaw" in name:
        return "agent-runtime"
    if "ensign" in name or "lora" in name:
        return "ensigns"
    if "research" in name or "experiment" in name:
        return "research"
    if "service" in name or "guard" in name:
        return "tools"
    if "memory" in name or "vision" in name:
        return "docs"
    if "old" in name or "deprecated" in name:
        return "legacy"
    
    # Default
    return "uncategorized"

def main():
    repos = fetch_repos()
    categorized = []
    for repo in repos[:20]:  # First 20 for this task
        cat = categorize(repo)
        categorized.append({
            "name": repo["name"],
            "url": repo["html_url"],
            "description": repo["description"],
            "topics": repo.get("topics", []),
            "category": cat,
        })
    
    # Output
    with open("work/2026-04-19_0810_categorized_repos.json", "w") as f:
        json.dump(categorized, f, indent=2)
    
    # Summary
    from collections import Counter
    counts = Counter([r["category"] for r in categorized])
    print("Categorized 20 repos:")
    for cat, count in counts.items():
        print(f"  {cat}: {count}")
    
    # Write markdown summary
    with open("work/2026-04-19_0810_categorized_repos.md", "w") as f:
        f.write("# Categorized Repos (20)\n\n")
        f.write("| Repo | Category | Description |\n")
        f.write("|------|----------|-------------|\n")
        for r in categorized:
            desc = (r["description"] or "")[:60]
            f.write(f"| [{r['name']}]({r['url']}) | {r['category']} | {desc} |\n")
        f.write(f"\n**Summary**: {dict(counts)}\n")

if __name__ == "__main__":
    main()
