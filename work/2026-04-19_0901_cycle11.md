GITHUB_USER = "SuperInstance"
API_URL = f"https://api.github.com/users/{GITHUB_USER}/repos"
HEADERS = {"Accept": "application/vnd.github.v3+json"}

# Fleet taxonomy categories with descriptions
TAXONOMY = {
    "core": "Essential infrastructure, foundational systems",
    "training": "Training rooms, simulators, LoRA tools",
    "agent": "Agent runtimes, shells, frameworks",
    "research": "Experiments, papers, prototypes",
    "tool": "Utilities, scripts, CLI tools",
    "archive": "Deprecated, inactive, historical",
    "docs": "Documentation, knowledge bases"
}

# Keywords mapping for automatic categorization
KEYWORD_MAP = {
    "core": ["plato", "holodeck", "fleet-simulator", "constraint-theory", "cudaclaw", "flux"],
    "training": ["train", "lora", "room", "ensign", "simulator", "bootcamp"],
    "agent": ["agent", "claw", "runtime", "shell", "zeroclaw"],
    "research": ["experiment", "prototype", "paper", "thesis", "study"],
    "tool": ["util", "script", "cli", "tool", "helper", "service"],
    "docs": ["docs", "wiki", "knowledge", "memory", "guide"],
    "archive": ["old", "deprecated", "legacy", "archive", "historical"]
}

def fetch_repos() -> List[Dict]:
    """Fetch all repos from SuperInstance GitHub."""
    repos = []
    page = 1
    
    while True:
        response = requests.get(f"{API_URL}?page={page}&per_page=100", headers=HEADERS)
        if response.status_code != 200:
            print(f"Error fetching repos: {response.status_code}")
            break
            
        page_repos = response.json()
        if not page_repos:
            break
            
        repos.extend(page_repos)
        page += 1
        
        # GitHub API pagination limit
        if len(page_repos) < 100:
            break
    
    return repos

def analyze_repo(repo: Dict) -> Dict:
    """Analyze a single repo for categorization."""
    name = repo.get("name", "")
    description = repo.get("description", "").lower() if repo.get("description") else ""
    topics = repo.get("topics", [])
    readme_content = fetch_readme(repo.get("name", ""))
    
    # Combine all text for analysis
    analysis_text = f"{name} {description} {' '.join(topics)} {readme_content}".lower()
    
    # Calculate category scores
    scores = {}
    for category, keywords in KEYWORD_MAP.items():
        score = 0
        for keyword in keywords:
            if keyword in analysis_text:
                score += 1
            if keyword in name.lower():
                score += 2  # Higher weight for name matches
        
        # Check for archive indicators
        if category == "archive":
            if repo.get("archived", False):
                score += 5
            if "deprecated" in analysis_text or "legacy" in analysis_text:
                score += 3
            if repo.get("updated_at", ""):
                # If not updated in over a year
                from datetime import datetime, timedelta
                try:
                    updated = datetime.fromisoformat(repo.get("updated_at").replace("Z", "+00:00"))
                    if datetime.now() - updated > timedelta(days=365):
                        score += 2
                except:
                    pass
        
        scores[category] = score
    
    # Determine primary category
    primary_category = max(scores.items(), key=lambda x: x[1])
    
    # Calculate confidence (0-100)
    total_score = sum(scores.values())
    confidence = int((primary_category[1] / max(total_score, 1)) * 100) if total_score > 0 else 0
    
    return {
        "name": name,
        "url": repo.get("html_url", ""),
        "description": repo.get("description", ""),
        "primary_category": primary_category[0],
        "confidence": confidence,
        "all_scores": scores,
        "archived": repo.get("archived", False),
        "updated": repo.get("updated_at", ""),
        "topics": topics
    }

def fetch_readme(repo_name: str) -> str:
    """Fetch README content for a repo."""
    readme_url = f"https://api.github.com/repos/{GITHUB_USER}/{repo_name}/readme"
    response = requests.get(readme_url, headers=HEADERS)
    
    if response.status_code == 200:
        content = response.json().get("content", "")
        # Decode base64
        import base64
        try:
            return base64.b64decode(content).decode('utf-8', errors='ignore').lower()
        except:
            return ""
    return ""

def categorize_repos() -> List[Dict]:
    """Main categorization function."""
    print(f"Fetching repos from {GITHUB_USER}...")
    repos = fetch_repos()
    
    print(f"Analyzing {len(repos)} repos...")
    categorized = []
    
    for repo in repos:
        analysis = analyze_repo(repo)
        categorized.append(analysis)
    
    # Sort by confidence descending
    categorized.sort(key=lambda x: x["confidence"], reverse=True)
    
    return categorized
