class GitHubRepoAnalyzer:
    """Analyzes GitHub repositories for categorization"""
    
    def __init__(self, username: str = "SuperInstance", token: Optional[str] = None):
        self.username = username
        self.token = token
        self.session = requests.Session()
        if token:
            self.session.headers.update({"Authorization": f"token {token}"})
        
    def fetch_repos(self) -> List[Dict]:
        """Fetch all repos for the user"""
        url = f"https://api.github.com/users/{self.username}/repos"
        repos = []
        page = 1
        
        while True:
            response = self.session.get(url, params={"page": page, "per_page": 100})
            response.raise_for_status()
            page_repos = response.json()
            if not page_repos:
                break
            repos.extend(page_repos)
            page += 1
            
        return repos
    
    def analyze_repo(self, repo: Dict) -> Dict:
        """Analyze a single repository"""
        repo_data = {
            "name": repo["name"],
            "description": repo["description"] or "",
            "url": repo["html_url"],
            "topics": repo.get("topics", []),
            "language": repo.get("language", ""),
            "updated_at": repo["updated_at"],
            "has_wiki": repo.get("has_wiki", False),
            "has_pages": repo.get("has_pages", False),
            "readme_content": "",
            "file_structure": [],
            "category_scores": {},
            "primary_category": "uncategorized"
        }
        
        # Try to fetch README
        readme_url = f"https://api.github.com/repos/{self.username}/{repo['name']}/readme"
        try:
            response = self.session.get(readme_url)
            if response.status_code == 200:
                readme_data = response.json()
                # Decode base64 content if present
                if "content" in readme_data:
                    import base64
                    repo_data["readme_content"] = base64.b64decode(
                        readme_data["content"]
                    ).decode('utf-8', errors='ignore')
        except:
            pass
        
        # Score against categories
        repo_text = f"{repo_data['description']} {repo_data['readme_content']}".lower()
        repo_text += " " + " ".join(repo_data["topics"]).lower()
        
        for category in FLEET_CATEGORIES:
            score = 0
            for keyword in category.keywords:
                if keyword.lower() in repo_text:
                    score += 1
            # Bonus for exact name matches
            if repo["name"] in category.examples:
                score += 3
            repo_data["category_scores"][category.name] = score
        
        # Determine primary category
        if repo_data["category_scores"]:
            primary = max(repo_data["category_scores"].items(), key=lambda x: x[1])
            if primary[1] > 0:
                repo_data["primary_category"] = primary[0]
        
        return repo_data
    
    def categorize_repos(self, limit: int = 20) -> List[Dict]:
        """Fetch and categorize repositories"""
        print(f"Fetching repos for {self.username}...")
        repos = self.fetch_repos()
        
        print(f"Analyzing {len(repos)} repositories...")
        categorized = []
        
        for i, repo in enumerate(repos[:limit]):
            print(f"  {i+1}/{min(len(repos), limit)}: {repo['name']}")
            categorized.append(self.analyze_repo(repo))
        
        return categorized
